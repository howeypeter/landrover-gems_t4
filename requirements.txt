# Runtime + test dependencies for gems_t4 — the CLI/protocol path, NOT the GUI.
#
# Quick install into a virtual environment:
#     python -m venv .venv
#     .\.venv\Scripts\pip install -r requirements.txt      # Windows
#     ./.venv/bin/pip install -r requirements.txt           # macOS / Linux
#
# Then run:  python -m gems_t4 scenarios
#
# PySide6 (the GUI) is deliberately NOT listed here — it's a large binary
# dependency (~200+ MB) that pure CLI/protocol work doesn't need. `pytest` with
# only this file installed gives "74 passed, 10 skipped" (the GUI test files
# skip cleanly via `pytest.importorskip("PySide6")`, not an error).
#
# Want the GUI too? Use pip's "extras" instead of this file:
#     pip install -e ".[gui]"      # adds PySide6 + registers the `gems_t4` command
#     pip install -e ".[dev]"      # adds PySide6 + pytest-qt, for running ALL tests
# Either gives "123 passed" from `pytest`.

pyserial>=3.5
rich>=13.0
pytest>=8.0
